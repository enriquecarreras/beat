#We define a function that creates new variables for the analysis
def process_data(data):
    
    data['bgc-signup'] = data['bgc_date'] - data['signup_timestamp']

    data['bgc-signup'] = data['bgc-signup'].dt.days

    data['vad-signup'] = data['vehicle_added_date'] - data['signup_timestamp']

    data['vad-signup'] = data['vad-signup'].dt.days

    data['bgc-signup'] = pd.cut(data['bgc-signup'], bins = 3)
    data['vad-signup'] = pd.cut(data['vad-signup'], bins = 3)


    data.loc[data['bgc_date'].isna(), 'has_bgc'] = 0
    data.loc[data['bgc_date'].notna(), 'has_bgc'] = 1

    data.loc[data['vehicle_added_date'].isna(), 'has_vad'] = 0
    data.loc[data['vehicle_added_date'].notna(), 'has_vad'] = 1
    
    data['top4_makers'] = data['vehicle_make'].apply(vehicles2)
    data['top20_models'] = data['vehicle_model'].apply(models2)
    data['yearbins'] = pd.cut(data['vehicle_year'], bins = 4)
    data['top5_years'] = data['vehicle_year'].apply(years2)
   
    data = data.drop(['id','signup_timestamp','bgc_date','vehicle_added_date','bgc-signup','vad-signup','first_completed_trip_timestamp','has_signup','vehicle_make','vehicle_model','first-signup'], axis = 1)
    return data

    p= []

    p = pd.Series(model.pvalues.values[1:],index = cols)      
    pmax = max(p)
    feature_with_p_max = p.idxmax()
    if(pmax>0.05):
        cols.remove(feature_with_p_max)
